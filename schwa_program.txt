var pi : float;
pi = 3.14159; 

if {
    p == 3 ~> abort
    p != 3 ~> skip
};

def myfunc ( radius : float ) : float 
{
    var flag : bool;
    var area : float;
    var perimeter: float;

    area = pi * radius*radius;
    perimeter = 2 * (radius + radius );
    flag = 1.0 ; 

    if {
        area < perimeter ~>
            do { area < perimiter ~> area = area + 1.0 }
        perimiter < area ~>
            flag = - flag
    }
};
